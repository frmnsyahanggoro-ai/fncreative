        const params = new URLSearchParams(window.location.search);
        const id = params.get('id');
        const display = document.getElementById('menu-name');
        const btn = document.getElementById('manual-link');

        if (id) {
            display.innerText = "Membuka Offer ID: " + id;
            
            // Kita coba 3 skema sekaligus
            const link1 = "mcdonalds://GMA/OfferDetails?offerId=" + id;
            const link2 = "globalmcdonalds://GMA/OfferDetails?offerId=" + id;
            const link3 = "https://u.mcd.com/app/offer-details?offerId=" + id;

            btn.href = link1;

            // Eksekusi otomatis
            window.location.href = link1;
            
            setTimeout(() => { window.location.href = link2; }, 800);
            setTimeout(() => { window.location.href = link3; }, 1500);
            
        } else {
            display.innerText = "Masukkan ID di URL (contoh: ?id=12345)";
            btn.style.display = "none";
        }
